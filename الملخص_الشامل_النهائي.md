#### إضافة 3: خوارزميات الحساب
```python
# 1. التضخم (معدل افتراضي 3.5% ← 8%)
inflation_rate = 3.5
if sel_days > 365:
    inflation_rate += (sel_days/365 - 1) * 0.8
else:
    inflation_rate *= (sel_days/365)

if sel_region in ["قطاع الرياض", "نيوم", ...]:
    inflation_rate *= 1.15

inflation_rate = round(max(0.5, min(inflation_rate, 8.0)), 2)

# 2. البصمة الكربونية (50 ← 800 طن)
carbon_base = {"صغير": 50, "متوسط": 150, ...}
carbon_emissions = carbon_base.get(sel_size, 100)
carbon_emissions *= (1.3 if sel_region_hot else 1.0)
carbon_emissions *= (sel_days / 180)
carbon_emissions *= (1.1 - sel_labor * 0.1)
carbon_footprint = round(carbon_emissions, 1)

# 3. الامتثال (10 نقاط، ≥6 = مطابق)
compliance_score = 0
if sel_labor > 0.85: compliance_score += 2
if budget_per_day > 50000: compliance_score += 2
# ... المعايير الأخرى
compliance_percentage = (compliance_score / 10) * 100
is_compliant = compliance_percentage >= 60
```
